python爬虫-博士学位分析-2

抓下来以后就是要分析了，先整理成数据库再进行数据处理。这部分等学会SQLite或者mysql后再具体进行。暂时使用一个简单的折衷方案：导入excel进行分析。
导入excel用到的是xlwt，后续的数据分析都在xlwt里进行。备注，xlwt是针对xls格式的，xlsx格式的需要用别的。

代码里面有个多余的二维数组m，是写在这里留着以后学会数据库操作后改的。

上面的代码可以比较的抓到大部分结果，但是总有一些意外，比如外文名、英文答辩标题等等，后来改成一个table一个table单独匹配，仍然只用到re。

# -*- coding: utf-8 -*-
"""
Created on Thu Aug 17 14:30:34 2017
@author: vocaliu
@version: Anaconda python3.6
"""

import os
import re
import xlwt

#初始化部分    
m=[]                                                   #建立一个二维数组，设置第一项相当于标题栏
n0=['来源','学号','答辩时间','姓名','院系','专业','论文题目','答辩地点','答辩秘书','主席','职称','单位','成员2','职称','单位','成员3','职称','单位','成员4','职称','单位','成员5','职称','单位','答辩秘书']
m.append(n0)                                          #把n0写进m里做m[0]

hangshu = 1                                           #m列表的第一行写成n0有个自制的标题行，实际数据量=行数-1；引用数据时编号从0开始，注意行数要减
new_workbook=xlwt.Workbook()                          #新建一个xls格式的excel并打开一个sheet
new_sheet=new_workbook.add_sheet("phd_data",cell_overwrite_ok=True)  
for i in range(24):
    new_sheet.write(0,i,m[0][i])                #把m的第一项(m[0])，逐个写入excel的第一行(行号为0)

#循环抓内容
for num in range (numstart,numstop):    
    filename = 'F:\phddbjl\{}.htm'.format(num)
    try:
        with open('{}'.format(filename),'rb') as fp:    #本地化
            data = fp.read()
            data = data.decode("gbk")               #改变编码从二进制变成显示中文
            data = repr(data)                       #字符化，有没有似乎没影响

#            pattern_xuehao = r'D[FGZ]\d{7}'             #df dg dz三种学号
#            pattern_dabianshijian = r'\d{4}\-\d{1,2}\-\d{1,2}'   
#为3个table分别重写pattern，t1是个人资料，t2是答辩委员会，t3是秘书时间地点
            pattern_t1 = r'</font></td><td height="\d{2}" width="\d{3}"><font size="\d{1}" color="#\d{6}">&nbsp;.*?\s*</font>'
            pattern_t2 = r'</td><td height="\d{2}" width="\d{3}" align="center"><font color="#\d{6}">&nbsp;.*?\s*</font>'
            pattern_t3 = r'</font></td><td height="\d{2}" width="\d{3}"><font color="#\d{6}">&nbsp;.*?\s*</font>'

            m_t1 = re.findall(pattern_t1,data)      
            m_t2 = re.findall(pattern_t2,data)         
            m_t3 = re.findall(pattern_t3,data)  

            n=[]                                  #拼一个list出来，再写入m作为二维list中的一项
            n.append(num)
            n.append(' '.join(m_t1[3][77:-7].split())) #1，选指定范围的内容；2，去掉多余的空格
            n.append(' '.join(m_t3[1][68:-7].split()))
            n.append(' '.join(m_t1[2][77:-7].split()))
            n.append(' '.join(m_t1[0][77:-7].split()))

            n.append(' '.join(m_t3[2][68:-7].split()))
            n.append(' '.join(m_t3[0][68:-7].split()))
            for x in range(15):
                n.append(' '.join(m_t2[x][76:-7].split()))            

            m.append(n)

        for i in range(24):
            new_sheet.write(hangshu,i,n[i])
