python爬虫-mmjpg

类似的抓了mmjpg的所有图

# -*- coding: utf-8 -*-
"""
Created on Fri Aug 18 03:20:14 2017
@author: vocaliu
@version: Anaconda python3.6
       网页         图片
2015:1-485，      1-489  
2016:486-850，   490-862    
2017:851-1079，  863-1080以上
"""

import urllib
import urllib.request
import re
import os
from multiprocessing import Pool
import time

def download_page(url):
    head = { 'User-Agent': 'Mozilla/5.0 (X11; Linux x86_64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/41.0.2272.89 Safari/537.36',  
             'Referer': 'http://www.mmjpg.com',  
             }                                         #伪装浏览器登陆
    request = urllib.request.Request(url,headers=head) #构建请求
    response = urllib.request.urlopen(request,timeout=10)
    data = response.read()                             #获得服务器响应并读取链接内容（网页、图片或其他）
    return data                                        #返回值就是链接中的具体内容

def imgdownload_num(year,startnum,stopnum):
    for num in range(startnum,stopnum):               #这个数字不断变化来抓不同链接
        pageurl = r'http://www.mmjpg.com/mm/' + str(num)            #图片页面

        wronglink = 0
        try:
            html_raw = download_page(pageurl)                  #在这里做一轮try except，防止遇到错误的抓取页面导致的报错
            html=html_raw.decode("UTF-8") 
            reg_title = r'<h2>.+?</h2>'                       #通过正则配对，找到页面标题
            reg_maxnum = r'>\d{1,3}</.{1,3}><em class='       #通过正则配对，得到页面中的图片的最大序号
            pattern_title = re.compile(reg_title)
            pattern_maxnum = re.compile(reg_maxnum)           #re.compile() 可以把正则表达式编译成一个正则表达式对象.
            t = re.findall(pattern_title,repr(html))
            max_num = re.findall(pattern_maxnum,repr(html))  #repr的作用参考http://www.cnpythoner.com/post/251.html
            title = t[0][4:-5]
            max_picnum = max_num[0][1:3]

        except:
            wronglink += 1
            print('发现第{}个错误的图片页面 -> '.format(wronglink,pageurl))#设置时会里有一些指向404页面的网址
            break

        if not os.path.exists('F:\pic\mmjpg\{}-{}'.format(num,title)):   #如果不存在与网址一样的文件夹
            os.mkdir('F:\pic\mmjpg\{}-{}'.format(num,title))             #新建与网址一样的文件夹

        wrongpiclink = 0
        for i in range(1,int(max_picnum)+1):
            picurl = r'http://img.mmjpg.com/' + str(year) + r'/' + str(num) + r'/' + str(i) +'.jpg'   #图片地址
            try:
                image = download_page(picurl)             #将每个img_link链接中的图片都读取出来，并保存成下面的本地链接
                with open('F:\pic\mmjpg\{}-{}\{}.jpg'.format(num,title,i),'wb') as fp:
                    fp.write(image)
                    print('pic{}-{} downloaded from {}'.format(num,i,picurl))
                    time.sleep(0.6)
            except:
                wrongpiclink += 1
                print('发现第{}个错误的图片链接 ---> {}'.format(wrongpiclink,picurl))#源文件里有一些指向404页面的图片

if __name__ == '__main__':
    print('Parent process %s.' % os.getpid())
    p = Pool(6)
    for i in range(6):
        p.apply_async(imgdownload_num, args=(2015,i*70+1,i*70+71))
    p.close()
    p.join()
    print('All subprocesses done.')
